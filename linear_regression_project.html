def matxMultiply(A, B):
    result_mp = []
    if len(A[0]) != len(B):
        raise ValueError
    for Ai in range(len(A)):
        result_row = []
        for Bj in range(len(B[0])):
            result = 0
            for Bi in range(len(B)):
                result += A[Ai][Bi] * B[Bi][Bj]
            result_row.append(result)
        result_mp.append(result_row)
    return result_mp

    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b):
        return None
    else:
        Ab = augmentMatrix(A, b)
        #print('Ab=', Ab)
    for j in range(len(Ab)):
        #print('j=', j)
        max_list = transpose(Ab)[j][j:]
        # list
        #print('max_list=', max_list)
        max_val = max(max_list, key = abs)
        #print('max_val=', max_val)
        # maxkeyGoogle
        if abs(max_val) < epsilon:
            return None
        else:
            idx = j + max_list.index(max_val)
            #print('idx=', idx)
        swapRows(Ab, j, idx)
        #print('Ab=', Ab)
        scaleRow(Ab, j, 1/Ab[j][j])
        #print('Ab=', Ab) 
        for i in range(len(Ab)):
            #print('i=', i)
            if i != j and Ab[i][j] != 0:
                addScaledRow(Ab, i, j, -Ab[i][j])
                #print('Ab=', Ab)
    result = [[row[-1]] for row in Ab]
    #print('result=', result)
    matxRound(result, decPts)
    #print('result=', result)
    return result
